Examples
Sample romloader-config.xml: 
        <configuration>
            <!-- trace logging on when true -->
            <logging enabled="true" />

            <!-- performance stats on when true -->
            <stats enabled="true" />

            <!-- rom to load, and its game id for scores storage -->
            <rom file="../controls/ControlTestGameRom.swf" id="com_pixeldroid_controltest" />

            <!-- key mappings, player numbers start at 1 -->
            <keymappings>
                <joystick playerNumber="1">
                    <hatRight keyCode="39" />
                    <hatUp    keyCode="38" />
                    <hatLeft  keyCode="37" />
                    <hatDown  keyCode="40" />
                    <buttonX  keyCode="17" />
                    <buttonA  keyCode="46" />
                    <buttonB  keyCode="35" />
                    <buttonC  keyCode="34" />
                </joystick>
            </keymappings>
            
            <!-- arbitrary property values -->
            <property name="fullScreen">true</property><!-- for DesktopRomLoader -->
            <property name="scoreServer">http://scores.foo.com</property><!-- for RC4D3RomLoader -->
            <property name="foo">foo value</property>
            <property name="bar">bar value</property>
            <property name="bat">bat value</property>

        </configuration>
