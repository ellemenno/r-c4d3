Classpublic class ConfigDataProxy
Implementscom.pixeldroid.r_c4d3.interfaces.IGameConfigProxy

ConfigDataProxy is a simple data structure front end to the romloader xml configuration.


Example
Sample romloader-config.xml: 
  <configuration>
   <!-- trace logging on when true -->
   <logging enabled="true" />

   <!-- rom to load, and its game id for scores storage -->
   <rom file="../controls/ControlTestGameRom.swf" id="com_pixeldroid_controltest" />

   <!-- key mappings, player numbers start at 1 -->
   <keymappings>
    <joystick playerNumber="1">
     <hatUp    keyCode="38" />
     <hatRight keyCode="39" />
     <hatLeft  keyCode="37" />
     <hatDown  keyCode="40" />
     <buttonY  keyCode="17" />
     <buttonR  keyCode="46" />
     <buttonG  keyCode="35" />
     <buttonB  keyCode="34" />
    </joystick>
   </keymappings>
   
   <!-- arbitrary property values -->
   <property name="scoreServer">http://scores.foo.com</property>
   <property name="foo">foo value</property>
   <property name="bar">bar value</property>
   <property name="bat">bat value</property>

  </configuration>
